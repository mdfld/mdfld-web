# Help Page Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a public `/help` page with tab pills, accordion Q&As, and client-side search across all sections.

**Architecture:** Static content lives in `lib/help-content.ts` as a typed config object. A single client component (`components/help/help-page.tsx`) owns all interactive state (active tab, open accordion items, search query). A thin server page wrapper at `app/(main)/help/page.tsx` renders it inside the existing main layout (navbar + footer). The dashboard sidebar "Help & Information" button gets wired to `/help`.

**Tech Stack:** Next.js 14 App Router, HeroUI (Accordion, Button), Tailwind CSS, Vitest

export function searchHelp(query: string): HelpSearchResult[] {
  const q = query.trim().toLowerCase();
  if (!q) return [];

  const results: HelpSearchResult[] = [];
  for (const section of HELP_CONTENT) {
    for (const item of section.questions) {
      if (
        item.q.toLowerCase().includes(q) ||
        item.a.toLowerCase().includes(q)
      ) {
        results.push({ ...item, sectionLabel: section.label });
      }
    }
  }
  return results;
}

- [ ] **Step 3: Create `components/help/help-page.tsx`**

```tsx
"use client";

import React, { useState } from "react";
import Link from "next/link";
import { Icon } from "@iconify/react";
import { HELP_CONTENT, searchHelp } from "@/lib/help-content";
import type { HelpSection } from "@/lib/help-content";

export function HelpPage() {
  const [activeTab, setActiveTab] = useState<string>(HELP_CONTENT[0].id);
  const [openItems, setOpenItems] = useState<Set<string>>(new Set());
  const [query, setQuery] = useState("");

  const isSearching = query.trim().length > 0;
  const searchResults = isSearching ? searchHelp(query) : [];
  const activeSection = HELP_CONTENT.find((s) => s.id === activeTab)!;

  function toggleItem(key: string) {
    setOpenItems((prev) => {
      const next = new Set(prev);
      next.has(key) ? next.delete(key) : next.add(key);
      return next;
    });
  }

  return (
    <div className="min-h-screen">
      {/* Hero */}
      <div className="border-b border-divider py-12 px-4 text-center">
        <h1 className="text-3xl font-bold mb-2">How can we help?</h1>
        <p className="text-default-500 text-sm mb-6">
          Answers for buyers, sellers, and everything in between
        </p>
        <div className="relative max-w-md mx-auto">
          <Icon
            icon="lucide:search"
            className="absolute left-3 top-1/2 -translate-y-1/2 text-default-400"
            width={16}
          />
          <input
            type="text"
            value={query}
            onChange={(e) => setQuery(e.target.value)}
            placeholder="Search help articles..."
            className="w-full pl-9 pr-4 py-2.5 rounded-lg bg-content2 border border-divider text-sm focus:outline-none focus:border-primary"
          />
          {query && (
            <button
              onClick={() => setQuery("")}
              className="absolute right-3 top-1/2 -translate-y-1/2 text-default-400 hover:text-foreground"
            >
              <Icon icon="lucide:x" width={14} />
            </button>
          )}
        </div>
      </div>

      <div className="max-w-2xl mx-auto px-4 py-8">
        {isSearching ? (
          <SearchResults query={query} results={searchResults} />
        ) : (
          <>
            {/* Tab pills */}
            <div className="flex gap-2 overflow-x-auto pb-2 mb-6 scrollbar-hide">
              {HELP_CONTENT.map((section) => (
                <button
                  key={section.id}
                  onClick={() => setActiveTab(section.id)}
                  className={`flex-shrink-0 px-4 py-1.5 rounded-full text-sm font-medium transition-colors ${
                    activeTab === section.id
                      ? "bg-primary text-primary-foreground"
                      : "border border-divider text-default-500 hover:text-foreground"
                  }`}
                >
                  {section.label}
                </button>
              ))}
            </div>

            {/* Accordion */}
            <Accordion
              section={activeSection}
              openItems={openItems}
              onToggle={toggleItem}
            />
          </>
        )}

        {/* Still need help */}
        <div className="mt-8 rounded-xl border border-divider bg-content1 p-5 flex flex-col sm:flex-row items-start sm:items-center justify-between gap-4">
          <div>
            <p className="font-semibold text-sm">Still need help?</p>
            <p className="text-default-500 text-xs mt-0.5">
              Our team usually responds within 24 hours.
            </p>
          </div>
          <Link
            href="/contact"
            className="text-sm font-medium bg-primary text-primary-foreground px-4 py-2 rounded-lg hover:opacity-90 transition-opacity whitespace-nowrap"
          >
            Contact us →
          </Link>
        </div>
      </div>
    </div>
  );
}

function Accordion({
  section,
  openItems,
  onToggle,
}: {
  section: HelpSection;
  openItems: Set<string>;
  onToggle: (key: string) => void;
}) {
  return (
    <div className="space-y-2">
      {section.questions.map((item, i) => {
        const key = `${section.id}-${i}`;
        const isOpen = openItems.has(key);
        return (
          <div
            key={key}
            className={`rounded-lg border transition-colors ${
              isOpen ? "border-primary" : "border-divider"
            }`}
          >
            <button
              onClick={() => onToggle(key)}
              className="w-full flex items-center justify-between px-4 py-3 text-left text-sm font-medium hover:text-primary transition-colors"
            >
              <span>{item.q}</span>
              <Icon
                icon={isOpen ? "lucide:minus" : "lucide:plus"}
                width={16}
                className={`flex-shrink-0 ml-3 ${isOpen ? "text-primary" : "text-default-400"}`}
              />
            </button>
            {isOpen && (
              <div className="px-4 pb-4 text-sm text-default-500 leading-relaxed border-t border-divider pt-3">
                {item.a}
              </div>
            )}
          </div>
        );
      })}
    </div>
  );
}

function SearchResults({
  query,
  results,
}: {
  query: string;
  results: ReturnType<typeof searchHelp>;
}) {
  if (results.length === 0) {
    return (
      <div className="text-center py-12 text-default-400 text-sm">
        No results for &ldquo;{query}&rdquo; — try a different term or{" "}
        <Link href="/contact" className="text-primary hover:underline">
          contact us
        </Link>
        .
      </div>
    );
  }

  // Group by sectionLabel
  const grouped = results.reduce<Record<string, typeof results>>((acc, r) => {
    if (!acc[r.sectionLabel]) acc[r.sectionLabel] = [];
    acc[r.sectionLabel].push(r);
    return acc;
  }, {});

  return (
    <div className="space-y-6">
      {Object.entries(grouped).map(([label, items]) => (
        <div key={label}>
          <p className="text-xs font-semibold uppercase tracking-wider text-default-400 mb-2">
            {label}
          </p>
          <div className="space-y-2">
            {items.map((item, i) => (
              <div
                key={i}
                className="rounded-lg border border-divider px-4 py-3"
              >
                <p className="text-sm font-medium mb-1">{item.q}</p>
                <p className="text-sm text-default-500 leading-relaxed">
                  {item.a}
                </p>
              </div>
            ))}
          </div>
        </div>
      ))}
    </div>
  );
}
```
